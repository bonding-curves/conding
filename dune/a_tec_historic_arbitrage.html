Combining The Data

df = pd.merge(honeyswap, tec_mint_burn, left_on='block_day', right_on='date', how='outer')
df['date'] = df['date'].fillna(df['block_day'])
df = df.sort_values('date')
df[['net_liq', 'total_supply']] = df[['net_liq', 'total_supply']].ffill().bfill()
df = df.fillna(0)
df = df.drop('block_day',axis=1)
df.hvplot.line(x='date', y=['net_liq','total_supply'])
